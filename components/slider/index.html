Slider

The Slider component allows users to quickly select a value within a range. They should be used when the upper and lower bounds to the range are invariable.

50%
import { Slider } from '@marigold/components';

Props

PropertyTypeDefaultDescription
disabledbooleanfalseIf true, the slider is disabled.
valuenumber[]The slider's value (controlled).
defaultValuenumber[]The slider's default value (uncontrolled).
maxValuenumber100The slider's maximum value.
stepnumberThe slider's step value.
formatOptionsIntl.NumberFormatOptions
widthstring100%Control the width of the field
...Yes you can use all regular attributes of input type = range!

Examples

Width

0
import { Slider } from '@marigold/components';


export const WidthSlider = () => (
  <Slider width="180px" aria-label="width">
    Example
  </Slider>
);

Disabled

0
import { Slider } from '@marigold/components';


export const DisabledSlider = () => <Slider disabled>Example</Slider>;

Step

0
import { Slider } from '@marigold/components';


export const StepSlider = () => <Slider step={10}>Example</Slider>;

Max value

0
import { Slider } from '@marigold/components';


export const MaxValueSlider = () => (
  <Slider aria-label="maxValue" maxValue={50}>
    Example
  </Slider>
);

Value (controlled)

75
import React, { useState } from 'react';
import { Slider } from '@marigold/components';


export const ControlledSlider = () => {
  const [value, setValue] = useState<number[] | number>(75);
  return (
    <Slider value={value} onChange={setValue}>
      Example
    </Slider>
  );
};

Default Value (uncontrolled)

25
import { Slider } from '@marigold/components';


export const UncontrolledSlider = () => (
  <Slider defaultValue={[25]}>Example</Slider>
);

Currency

€0.00
import { Slider } from '@marigold/components';


export const CurrencySlider = () => (
  <Slider formatOptions={{ style: 'currency', currency: 'EUR' }}>Price</Slider>
);

0%
import { Slider } from '@marigold/components';


export const PercentageSlider = () => (
  <Slider formatOptions={{ style: 'percent' }} step={0.01} maxValue={1}>
    Percent
  </Slider>
);
